function doPost(e) { try { var data = JSON.parse(e.postData.contents); // 1. IDENTIFICATION DU TABLEAU var ss = SpreadsheetApp.openById("1s890CM3o2X7Hv_QGggmO7pjMnBdmdss59WMEvcjcEi8"); // 2. CIBLAGE PRÉCIS DE L'ONGLET var sheet = ss.getSheetByName("Collecte_tests_morpho"); if (!sheet) { // Sécurité : si l'onglet est mal nommé, on crée une erreur visible dans les logs throw new Error("L'onglet 'Collecte_tests_morpho' est introuvable."); } // 3. IDENTIFICATION DU DOSSIER DRIVE var folder = DriveApp.getFolderById("1Z84CydtYoMR5XTVJ8k2nfNTZ1h54X90F"); var labels = ["Face", "Profil", "Dos"]; var photos = [data.photoFace, data.photoProfil, data.photoDos]; var urls = []; // 4. ENREGISTREMENT DES PHOTOS for (var i = 0; i < 3; i++) { var contentType = photos[i].split(",")[0].split(":")[1].split(";")[0]; var bytes = Utilities.base64Decode(photos[i].split(",")[1]); var fileName = "IDTEST_" + data.id_testeur + "_" + labels[i] + ".jpg"; var blob = Utilities.newBlob(bytes, contentType, fileName); var file = folder.createFile(blob); // Accès public pour la lecture par l'IA file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); urls.push(file.getUrl()); } // 5. AJOUT DE LA LIGNE DANS LE TABLEAU // Ordre : ID Testeur | Date | Stature | Photo Face | Photo Profil | Photo Dos sheet.appendRow([ data.id_testeur, new Date(), data.stature, urls[0], urls[1], urls[2] ]); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService.createTextOutput("Erreur: " + err.toString()).setMimeType(ContentService.MimeType.TEXT); } }
